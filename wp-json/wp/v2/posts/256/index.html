\ngladheid & ontwarring
\nin plaats van blond & volume
\ndakantochni<\/p>\n

dit komt nooit meer goed<\/p>\n

slaat ze me daar toch een figuur
\noksels niet geschoren
\neen nagelriem gescheurd
\ndakantochni<\/p>\n

\nik zeg nog doe omega drie

\ndraait ze een bonte was op zestig
\nzonder voorprogramma
\ndakantochni<\/p>\n
